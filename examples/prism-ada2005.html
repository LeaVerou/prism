To use this language, use the class "language-ada2005".

Strings

"foo ""bar"" baz"
"Multi-line strings are appended with a " &
"ampersand symbole."

Full example

package A is
   type I is new interface;
end A;

with Ada.Text_IO; Use Ada.Text_IO;

--  Comments look like this.
procedure Test is
   procedure Bah;
   pragma Import (C, "Bah", "bah");

   type Things is range 1 .. 10;
begin
   Put_Line ("Hello");   --  Comments look like this.

   Bah;  -- Call C function.

   for Index in Things'Range loop
      null;
   end loop;
end Test;
