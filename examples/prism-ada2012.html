Ada 2012

To use this language, use the class "language-ada2012".

Strings

"foo ""bar"" baz"
"Multi-line strings are appended with a " &
"ampersand symbole."

Full example

with Ada.Text_IO; Use Ada.Text_IO;

--  Comments look like this.
procedure Test is
   procedure Bah with
    Import        => True,   --  Shows the new aspect feature of the language.
    Convention    => C,
    External_Name => "bah";

   type Things is range 1 .. 10;
begin
   Put_Line ("Hello");   --  Comments look like this.

   Bah;  -- Call C function.

   for Index in Things'Range loop
      null;
   end loop;
end Test;
